FROM : to pull the base image

RUN : to execute commands

CMD: to provide defaults for an executing containers

ENTRYPOINT:to configure a container that will run as executable

WORKDIR:to sets the working directory

COPY:to copy a directory from your local machine to thedocker contant runtime
 
ADD:to copy files and folders from your local machine to docker containers

EXPOSE:informs docker that the container listens on the specified network ports

ENV:to set environment variables
